--CRIAÇÃO DE SEQUENCES E TABELAS

CREATE SEQUENCE SEQ_DADOS_IBGE
    INCREMENT 1
    MINVALUE 1 
    MAXVALUE 99999999999
    START 1
    CACHE 1 
;

CREATE SEQUENCE SEQ_MUNICIPIO
    INCREMENT 1
    MINVALUE 1 
    MAXVALUE 99999999999
    START 1
    CACHE 1 
;

CREATE TABLE MUNICIPIO (
    ID_MUNICIPIO INTEGER DEFAULT NEXTVAL('SEQ_MUNICIPIO'),
    NOME VARCHAR(120),
    REGIAO VARCHAR(50),
    UF CHAR(2),
    DESCRICAO_UF VARCHAR(50),
    PRIMARY KEY(ID_MUNICIPIO) 
);

SELECT * FROM MUNICIPIO;


CREATE TABLE DADOS_IBGE (
    ID_DADOS_IBGE INTEGER DEFAULT NEXTVAL('SEQ_DADOS_IBGE'),
    ID_MUNICIPIO INTEGER REFERENCES MUNICIPIO(ID_MUNICIPIO) NOT NULL,
    COD_IBGE NUMERIC,
    COD_IBGE7 NUMERIC,
    POPULACAO_2010 NUMERIC, 
    PORTE VARCHAR(50),
    IS_CAPITAL NUMERIC(1) DEFAULT 0,
    PRIMARY KEY(ID_DADOS_IBGE) 
);

SELECT  * FROM DADOS_IBGE;
